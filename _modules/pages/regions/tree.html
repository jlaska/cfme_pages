[docs]    def find_node_by_regexp(self, regexp_str):
        # finds first node by name in the whole tree, breadth first
        regexp = re.compile(regexp_str)
        if regexp.match(self.name):
            return self
        self.twisty.expand()
        queue = self.children
        while queue:
            child = queue.pop(0)
            if regexp.match(child.name):
                return child
            else:
                child.twisty.expand()
                queue += child.children
